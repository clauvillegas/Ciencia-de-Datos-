# Pega aquí el código principal de "Python-Clases.ipynb"
# Ejemplo mínimo:
from dataclasses import dataclass

@dataclass
class Producto:
    nombre: str
    precio: float

    def con_descuento(self, pct: float) -> float:
        return round(self.precio * (1 - pct), 2)

import pandas as pd

def resumen(df: pd.DataFrame):
    return {
        "filas": len(df),
        "columnas": len(df.columns),
        "nulos": int(df.isna().sum().sum()),
    }

class Biblioteca:
    def __init__(self):
        self.prestamos = {}

    def prestar(self, usuario, libro):
        self.prestamos.setdefault(usuario, []).append(libro)

# Pega aquí el código principal de "Python-Iteracion.ipynb"
# Ejemplo mínimo:
def take(n, iterable):
    for i, x in enumerate(iterable):
        if i == n: break
        yield x
