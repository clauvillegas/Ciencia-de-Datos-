from dataclasses import dataclass

@dataclass
class Producto:
    nombre: str
    precio: float

    def con_descuento(self, pct: float) -> float:
        return round(self.precio * (1 - pct), 2)

Geometría

Figuras, áreas y perímetros. Fuente: Python-Geometria.ipynb


def calcular_area_rectangulo(base, altura):
    area = base * altura
    return area

def calcular_circulo(radio):
    import math
    area = math.pi * radio ** 2
    return area

def calcular_promedio(numeros):
    suma = sum(numeros)
    promedio = suma / len(numeros)
    return promedio

def calcular_promedio_avanzado(numeros):
    import statistics
    promedio = statistics.mean(numeros)
    return promedio


def generar_numeros_aleatorios(cantidad, limite):
    import random
    numeros_aleatorios = [random.randint(1, limite) for _ in range(cantidad)]
    return numeros_aleatorios

def validanumero(tipo,mensaje):
    while True:
        try:
            if tipo == "int":
              numero = int(input(mensaje))
            elif tipo == "float":
              numero = float(input(mensaje))
            break  # Salir del bucle si el input es válido
        except ValueError:
            print("Error: Debes ingresar un número válido.")
    return numero

def resumen(df: pd.DataFrame):
    return {
        "filas": len(df),
        "columnas": len(df.columns),
        "nulos": int(df.isna().sum().sum()),

class Biblioteca:
    def __init__(self):
        self.prestamos = {}

    def prestar(self, usuario, libro):
        self.prestamos.setdefault(usuario, []).append(libro)

def take(n, iterable):
    for i, x in enumerate(iterable):
        if i == n: break
        yield x
